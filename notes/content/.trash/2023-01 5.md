dv.span("**💊 Pills**")

const calendarData = {
    year: 2023,
    entries: []
}

let tasksPerPage = dv.pages('"Agenda/Daily"').map((page) => {
	return page.file.tasks
})

where(p=>p.filte.tasks)
// .file.tasks.where(t => t.text.includes("#tag"))

for(let page of dv.pages('"Agenda/Daily"').where(p=>p.pills)){
    calendarData.entries.push({
        date: page.file.name,
        
    })
       
}

renderHeatmapCalendar(this.container, calendarData)
